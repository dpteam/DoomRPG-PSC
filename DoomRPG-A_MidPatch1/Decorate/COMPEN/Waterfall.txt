Actor WaterParticle
{
  Height 0
  Radius 0
  Mass 0
  +Missile
  +NoBlockMap
  -NoGravity
  +DontSplash
  +DoomBounce
  +FORCEXYBILLBOARD
  +CLIENTSIDEONLY
  BounceFactor 0.1
  Gravity 0.3
  RenderStyle Add
  Scale 2.2
  Speed 5
  States
  {
  Spawn:
    WPR1 A 1
    Goto Stand
  Stand:
    WPR1 A 2 A_FadeOut(0.03)
    Loop
  Death:
    Stop
  }
}

Actor WaterfallParticle
{
  Game Doom
  Height 0
  Radius 0
  Mass 0
  +Missile
  +NoBlockMap
  -NoGravity
  +DontSplash
  +FORCEXYBILLBOARD
  +CLIENTSIDEONLY
  BounceFactor 0.1
  Gravity 0.9
  RenderStyle Translucent
  Alpha 0.5
  Scale 2.0
  Speed 3
  States
  {
  Death:
    TNT1 A 0 A_CustomMissile ("WaterParticle", 6, 0, random (0, 360), 2, random (70, 110))
    Stop
  Spawn:
	WPR1 B 1 BRIGHT A_JumpIf(waterlevel > 0, "Death")
    Loop
  }
}
