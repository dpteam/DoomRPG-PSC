actor StrainDemon 20011 {
	hitobituary "%o was bit by a demon."
	health 217
	painchance 180
	speed 10
	radius 30
	height 56
	mass 400
	seesound "demon/sight"
	attacksound "demon/melee"
	painsound "demon/pain"
	deathsound "demon/death"
	activesound "demon/active"
	MONSTER
	+FLOORCLIP
	states {
		Spawn:
			SAR2 AB 10 A_Look
			loop
		See:
			SAR2 AABBCCDD 2 A_Chase
			loop
		Melee:
			SAR2 EF 8 A_FaceTarget
			SAR2 G 8 A_SargAttack
			goto See
		Pain:
			SAR2 H 2
			SAR2 H 2 A_Pain
			goto See
		Death:
			SAR2 I 8
			SAR2 J 8 A_Scream
			SAR2 K 4
			SAR2 L 4 A_Fall
			SAR2 M 4
			SAR2 N -1
			stop
		Raise:
			SAR2 NMLKJI 5
			goto See
	}
}
